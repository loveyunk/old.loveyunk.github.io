{% highlight html%}
<!doctype html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport"
          content="width=device-width, user-scalable=no, initial-scale=1.0, maximum-scale=1.0, minimum-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>Document</title>
    <style>
        .con {
            height: 1000px;
        }

        .wrapper p {
            height: 500px;
            border-top: 2px solid orange;
            opacity: 0;
            animation: move 1s forwards; /* 开始动画并停留在最后一帧 */
            animation-play-state: paused; /* 一开始动画暂停的 */
        }

        .wrapper p.move {
            animation-play-state: running; /* 开始动画 */
        }

        @keyframes move {
            from {
                opacity: 0;
                margin-left: 500px;
            }
            to {
                opacity: 1;
                margin-left: 0;
            }
        }
    </style>
</head>
<body>
<div class="con">文字文字文字文字文字文字文字</div>
<div class="wrapper">
    <hr>
    <p>第一屏进入视线</p>
    <hr>
    <p>第二屏进入视线</p>
    <hr>
    <p>第三屏进入视线</p>
</div>

<script src="jquery-3.2.1.js"></script>
<script>

    $(function () {
        let clientHeight = $(window).height(); // 浏览器视口高度
        $(window).scroll(function () {
            let currentScrollTop = $(this).scrollTop();
            $('p').each(function () {
                let targetTop = $(this).offset().top;
                if (clientHeight + currentScrollTop - 400 > targetTop) { // 目标区域进入视线, 400可指定距底部多少距离处开始动画
                    $(this).addClass('move');
                } else {
                    $(this).removeClass('move');
                }
            });
        });

    });

</script>
</body>
</html>
